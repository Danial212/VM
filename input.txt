LOAD R9 1
INP 20 #100
INP 20 #200
LOAD #10 100
LOAD #20 200

LABEL compare_loop

LOAD R3 [#10]
LOAD R4 [#20]
OUT -s #100
OUT -s #200
IF R3 == -1 GOTO check_end
IF R4 == -1 GOTO not_equal
IF R3 != R4 GOTO not_equal

LOAD R5 #10
LOAD R6 #20

ADD R5 R9
ADD R6 R9

LOAD #10 R5
LOAD #20 R6
GOTO compare_loop

LABEL check_end
IF R4 == -1 GOTO equal
GOTO not_equal

LABEL not_equal
OUT 0
GOTO END

LABEL equal
OUT 1
LABEL END
